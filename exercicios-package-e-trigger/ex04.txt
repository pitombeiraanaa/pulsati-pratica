CREATE OR REPLACE PACKAGE ANA_PKG_2 AS

FUNCTION RETORNA_PAIS(ID_ESTADO NUMBER) RETURN VARCHAR2;
END ANA_PKG_2;

CREATE OR REPLACE PACKAGE BODY ANA_PKG_2 AS
FUNCTION RETORNA_PAIS(ID_ESTADO NUMBER) RETURN VARCHAR2 
IS
PAIS VARCHAR2(100) := '';
BEGIN
    SELECT NOME_PAIS INTO PAIS FROM ADRI_PAISES
    INNER JOIN ana_estados ON ana_estados.id_pais = ana_paises.pais_id
    WHERE ESTADO_ID = ID_ESTADO;
    RETURN PAIS;
END RETORNA_PAIS;
END ANA_PKG_2;

DECLARE
RESULTADO VARCHAR2(100);

BEGIN
    RESULTADO:= ANA_PKG_2.RETORNA_PAIS(1);
    dbms_output.put_line(RESULTADO);
END;
